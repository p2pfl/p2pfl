19"""Flax Dataset export strategy.""" 

20 

21from typing import Any, Callable, Generator, Optional, Tuple 

22 

23import jax.numpy as jnp 

24from datasets import Dataset # type: ignore 

25from torch.utils.data import DataLoader 

26 

27from p2pfl.learning.dataset.p2pfl_dataset import DataExportStrategy 

28 

29 

30class FlaxExportStrategy(DataExportStrategy): 

31 """Export strategy for JAX/Flax datasets.""" 

32 

33 @staticmethod 

34 def export( 

35 data: Dataset, 

36 transforms: Optional[Callable] = None, 

37 batch_size: int = 1, 

38 num_workers: int = 0, 

39 **kwargs, 

40 ) -> Generator[Tuple[jnp.ndarray, jnp.ndarray], Any, None]: 

41 """ 

42 Export the data using the PyTorch strategy. 

43 

44 Args: 

45 data: The data to export. 

46 transforms: The transforms to apply to the data. 

47 batch_size: The batch size to use for the exported data. 

48 num_workers: The number of workers to use for the exported 

49 kwargs: Additional keyword arguments. 

50 

51 Returns: 

52 The exported data. 

53 

54 """ 

55 if transforms is not None: 

56 raise NotImplementedError("Transforms are not supported in this export strategy.") 

57 

58 # TODO: fix dataloader .with_format(type="jax", ...) with custom collate_fn 

59 torch_loader = DataLoader(data.with_format(type="torch", output_all_columns=True), batch_size=batch_size, num_workers=num_workers) 

60 

61 def jax_batch_generator() -> Generator[Tuple[jnp.ndarray, jnp.ndarray], Any, None]: 

62 for batch in torch_loader: 

63 features = jnp.array(batch["image"].numpy()) 

64 labels = jnp.array(batch["label"].numpy()) 

65 yield features, labels 

66 

67 return jax_batch_generator()
