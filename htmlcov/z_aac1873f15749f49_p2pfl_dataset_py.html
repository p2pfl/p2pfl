55class P2PFLDataset: 

56 """ 

57 Handle various data sources for Peer-to-Peer Federated Learning (P2PFL). 

58 

59 This class uses Hugging Face's `datasets.Dataset` as the intermediate representation for its flexibility and 

60 optimizations. 

61 

62 Supported data sources: 

63 - CSV files 

64 - JSON files 

65 - Parquet files 

66 - Python dictionaries 

67 - Python lists 

68 - Pandas DataFrames 

69 - Hugging Face datasets 

70 - SQL databases 

71 

72 To load different data sources, it is recommended to directly instantiate the `datasets.Dataset` object 

73 and pass it to the `P2PFLDataset` constructor. 

74 

75 Example: 

76 Load data from various sources and create a `P2PFLDataset` object: 

77 

78 .. code-block:: python 

79 

80 from datasets import load_dataset, DatasetDict, concatenate_datasets 

81 

82 # Load data from a CSV file 

83 dataset_csv = load_dataset("csv", data_files="data.csv") 

84 

85 # Load from the Hub 

86 dataset_hub = load_dataset("squad", split="train") 

87 

88 # Create the final dataset object 

89 p2pfl_dataset = P2PFLDataset( 

90 DatasetDict({ 

91 "train": concatenate_datasets([dataset_csv, dataset_hub]), 

92 "test": dataset_json 

93 }) 

94 ) 

95 

96 .. todo:: 

97 Add more complex integrations (databricks, etc.) 

98 

99 """ 

100 

101 def __init__( 

102 self, 

103 data: Union[Dataset, DatasetDict], 

104 train_split_name: str = "train", 

105 test_split_name: str = "test", 

106 batch_size: int = 1, 

107 ): 

108 """ 

109 Initialize the P2PFLDataset object. 

110 

111 Args: 

112 data: The dataset to use. 

113 train_split_name: The name of the training split. 

114 test_split_name: The name of the test split. 

115 batch_size: The batch size for the dataset. 

116 

117 """ 

118 self._data = data 

119 self._train_split_name = train_split_name 

120 self._test_split_name = test_split_name 

121 self.batch_size = batch_size 

122 

123 def get(self, idx, train: bool = True) -> Dict[str, Any]: 

124 """ 

125 Get the item at the given index. 

126 

127 Args: 

128 idx: The index of the item to retrieve. 

129 train: If True, get the item from the training split. Otherwise, get the item from the test split. 

130 

131 Returns: 

132 The item at the given index. 

133 

134 """ 

135 if isinstance(self._data, Dataset): 

136 data = self._data[idx] 

137 elif isinstance(self._data, DatasetDict): 

138 split = self._train_split_name if train else self._test_split_name 

139 data = self._data[split][idx] 

140 return data 

141 

142 def set_transforms(self, transforms: Callable) -> None: 

143 """ 

144 Set the transforms to apply to the data, delegating to the Hugging Face dataset. 

145 

146 Args: 

147 transforms: The transforms to apply to the data. 

148 

149 """ 

150 if isinstance(self._data, Dataset): 

151 self._data.set_transform(transforms) 

152 elif isinstance(self._data, DatasetDict): 

153 for split in self._data: 

154 self._data[split].set_transform(transforms) 

155 

156 def set_batch_size(self, batch_size: int) -> None: 

157 """ 

158 Set the batch size for the dataset. 

159 

160 Args: 

161 batch_size: The batch size for the dataset. 

162 

163 """ 

164 self.batch_size = batch_size 

165 

166 def generate_train_test_split(self, **kwargs) -> None: 

167 """ 

168 Generate a train/test split of the dataset. 

169 

170 Args: 

171 **kwargs: Additional keyword arguments to pass to the train_test_split method. 

172 

173 """ 

174 if isinstance(self._data, Dataset): 

175 self._data = self._data.train_test_split(**kwargs) 

176 elif isinstance(self._data, DatasetDict): 

177 if self._train_split_name in self._data and self._test_split_name in self._data: 

178 raise ValueError("Train and test splits already exist. Use a different name or clear the dataset.") 

179 train_test_split = self._data[self._train_split_name].train_test_split(**kwargs) 

180 self._data = DatasetDict( 

181 { 

182 self._train_split_name: train_test_split[self._train_split_name], 

183 self._test_split_name: train_test_split[self._test_split_name], 

184 } 

185 ) 

186 else: 

187 raise ValueError("Unsupported data type.") 

188 

189 def get_num_samples(self, train: bool = True) -> int: 

190 """ 

191 Get the number of samples in the dataset. 

192 

193 Args: 

194 train: If True, get the number of samples in the training split. Otherwise, get the number of samples in the test split. 

195 

196 Returns: 

197 The number of samples in the dataset. 

198 

199 """ 

200 if isinstance(self._data, Dataset): 

201 return len(self._data) 

202 elif isinstance(self._data, DatasetDict): 

203 split = self._train_split_name if train else self._test_split_name 

204 return len(self._data[split]) 

205 else: 

206 raise TypeError("Unsupported data type.") 

207 

208 def generate_partitions( 

209 self, num_partitions: int, strategy: DataPartitionStrategy, seed: int = 666, label_tag: str = "label", **kwargs 

210 ) -> List["P2PFLDataset"]: 

211 """ 

212 Generate partitions of the dataset. 

213 

214 Args: 

215 num_partitions: The number of partitions to generate. 

216 strategy: The partition strategy to use. 

217 seed: The random seed to use for reproducibility. 

218 label_tag: The tag to use for the label. 

219 **kwargs: Additional keyword arguments for the partition strategy. 

220 

221 Returns: 

222 An iterable of P2PFLDataset objects. 

223 

224 """ 

225 if isinstance(self._data, Dataset): 

226 raise ValueError("Cannot generate partitions for single datasets. ") 

227 train_partition_idxs, test_partition_idxs = strategy.generate_partitions( 

228 self._data[self._train_split_name], 

229 self._data[self._test_split_name], 

230 num_partitions, 

231 seed=seed, 

232 label_tag=label_tag, 

233 **kwargs, 

234 ) 

235 return [ 

236 P2PFLDataset( 

237 DatasetDict( 

238 { 

239 self._train_split_name: self._data[self._train_split_name].select(train_partition_idxs[i]), 

240 self._test_split_name: self._data[self._test_split_name].select(test_partition_idxs[i]), 

241 } 

242 ), 

243 train_split_name=self._train_split_name, 

244 test_split_name=self._test_split_name, 

245 batch_size=self.batch_size, 

246 ) 

247 for i in range(num_partitions) 

248 ] 

249 

250 def export( 

251 self, 

252 strategy: Type[DataExportStrategy], 

253 train: bool = True, 

254 **kwargs, 

255 ) -> Any: 

256 """ 

257 Export the dataset using the given strategy. 

258 

259 Args: 

260 strategy: The export strategy to use. 

261 train: If True, export the training data. Otherwise, export the test data. 

262 **kwargs: Additional keyword arguments for the export strategy. 

263 

264 Returns: 

265 The exported data. 

266 

267 """ 

268 # Checks 

269 if isinstance(self._data, Dataset): 

270 raise ValueError("Cannot export single datasets. Need to generate train/test splits first.") 

271 

272 # Export 

273 split = self._train_split_name if train else self._test_split_name 

274 return strategy.export(self._data[split], batch_size=self.batch_size, **kwargs)
