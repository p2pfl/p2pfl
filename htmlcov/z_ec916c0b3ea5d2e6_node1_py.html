19""" 

20Example of a P2PFL MNIST node using a MLP model and a MnistFederatedDM. 

21 

22This node only starts, create a node2 and connect to it in order to start the federated learning process. 

23""" 

24 

25import argparse 

26 

27from p2pfl.learning.dataset.p2pfl_dataset import P2PFLDataset 

28from p2pfl.learning.frameworks.pytorch.lightning_learner import LightningLearner 

29from p2pfl.learning.frameworks.pytorch.lightning_model import MLP, LightningModel 

30from p2pfl.node import Node 

31from p2pfl.utils.utils import set_test_settings 

32 

33set_test_settings() 

34 

35 

36def __get_args() -> argparse.Namespace: 

37 parser = argparse.ArgumentParser(description="P2PFL MNIST node using a MLP model and a MnistFederatedDM.") 

38 parser.add_argument("--port", type=int, help="The port.", required=True) 

39 return parser.parse_args() 

40 

41 

42def node1(port: int) -> None: 

43 """ 

44 Start a node1 and waits for a key press to stop it. 

45 

46 Args: 

47 port: The port where the node will be listening. 

48 

49 """ 

50 node = Node(LightningModel(MLP()), P2PFLDataset.from_huggingface("p2pfl/MNIST"), address=f"127.0.0.1:{port}", learner=LightningLearner) 

51 node.start() 

52 

53 input("Press any key to stop\n") 

54 

55 node.stop() 

56 

57 

58if __name__ == "__main__": 

59 args = __get_args() 

60 node1(args.port)
