35 """

107 def handshake(self, request: node_pb2.HandShakeRequest, _: grpc.ServicerContext) -> node_pb2.ResponseMessage: 

108 """ 

109 Service. It is called when a node connects to another. 

110 

111 Args: 

112 request: Request message. 

113 _: Context. 

114 

115 """ 

116 if self._neighbors.add(request.addr, non_direct=False, handshake=False): 

117 return node_pb2.ResponseMessage() 

118 else: 

119 return node_pb2.ResponseMessage(error="Cannot add the node (duplicated or wrong direction)") 

120 

121 def disconnect(self, request: node_pb2.HandShakeRequest, _: grpc.ServicerContext) -> google.protobuf.empty_pb2.Empty: 

122 """ 

123 Service. It is called when a node disconnects from another. 

124 

125 Args: 

126 request: Request message. 

127 _: Context. 

128 

129 """ 

130 self._neighbors.remove(request.addr, disconnect_msg=False) 

131 return google.protobuf.empty_pb2.Empty() 

132 

133 def send(self, request: node_pb2.RootMessage, _: grpc.ServicerContext) -> node_pb2.ResponseMessage: 

134 """ 

135 Service. Handles both regular messages and model weights. 

136 

137 Args: 

138 request: The RootMessage containing either a Message or Weights payload. 

139 _: Context. 

140 

141 """ 

142 # If message already processed, return 

143 if request.HasField("gossip_message") and not self._gossiper.check_and_set_processed(request): 

144 return node_pb2.ResponseMessage() 

145 

146 # Log 

147 package_type = "message" if request.HasField("gossip_message") else "weights" 

148 package_size = len(request.SerializeToString()) 

149 # Pass None for negative rounds, the logger will handle it 

150 round_num = request.round if request.round >= 0 else None 

151 logger.log_communication( 

152 self.addr, 

153 "received", 

154 request.cmd, 

155 request.source, 

156 package_type, 

157 package_size, 

158 round_num, 

159 ) 

160 

161 # Process message/model 

162 cmd_out: str | None = None 

163 if request.cmd in self.__commands: 

164 try: 

165 if request.HasField("gossip_message"): 

166 cmd_out = self.__commands[request.cmd].execute(request.source, request.round, *request.gossip_message.args) 

167 elif request.HasField("direct_message"): 

168 cmd_out = self.__commands[request.cmd].execute(request.source, request.round, *request.direct_message.args) 

169 elif request.HasField("weights"): 

170 cmd_out = self.__commands[request.cmd].execute( 

171 request.source, 

172 request.round, 

173 weights=request.weights.weights, 

174 contributors=request.weights.contributors, 

175 num_samples=request.weights.num_samples, 

176 ) 

177 else: 

178 error_text = f"Error while processing command: {request.cmd}: No message or weights." 

179 logger.error(self.addr, error_text) 

180 return node_pb2.ResponseMessage(error=error_text) 

181 except Exception as e: 

182 error_text = f"Error while processing command: {request.cmd}. {type(e).__name__}: {e}" 

183 logger.error(self.addr, error_text + f"\n{traceback.format_exc()}") 

184 return node_pb2.ResponseMessage(error=error_text) 

185 else: 

186 # disconnect node 

187 logger.error(self.addr, f"Unknown command: {request.cmd} from {request.source}") 

188 return node_pb2.ResponseMessage(error=f"Unknown command: {request.cmd}") 

189 

190 # If message gossip 

191 if request.HasField("gossip_message") and request.gossip_message.ttl > 0: 

192 # Update ttl and gossip 

193 request.gossip_message.ttl -= 1 

194 self._gossiper.add_message(request) 

195 

196 return node_pb2.ResponseMessage(response=cmd_out)

204 """

206 

207 Args:

215 self.__commands[cmds.get_name()] = cmds 

216 else: 

217 raise Exception("Command not valid")
