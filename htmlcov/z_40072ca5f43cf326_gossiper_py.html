36class Gossiper(threading.Thread, NodeComponent): 

37 """Gossiper for agnostic communication protocol.""" 

38 

39 def __init__( 

40 self, 

41 neighbors: Neighbors, 

42 build_msg: Callable[..., node_pb2.RootMessage], 

43 period: float | None = None, 

44 messages_per_period: int | None = None, 

45 ) -> None: 

46 """Initialize the gossiper.""" 

47 if period is None: 

48 period = Settings.gossip.PERIOD 

49 if messages_per_period is None: 

50 messages_per_period = Settings.gossip.MESSAGES_PER_PERIOD 

51 # Thread 

52 super().__init__() 

53 self.name = "gossiper-thread-unknown" 

54 

55 # Lists, locks and flag 

56 self.__processed_messages: list[int] = [] 

57 self.__processed_messages_lock = threading.Lock() 

58 self.__pending_msgs: list[tuple[node_pb2.RootMessage, list[ProtobuffClient]]] = [] 

59 self.__pending_msgs_lock = threading.Lock() 

60 self.__gossip_terminate_flag = threading.Event() 

61 

62 # Props 

63 self.__neighbors = neighbors 

64 self.period = period 

65 self.messages_per_period = messages_per_period 

66 

67 # Build msgs 

68 self.build_msg_fn = build_msg 

69 

70 def set_addr(self, addr: str) -> str: 

71 """Set the address.""" 

72 addr = super().set_addr(addr) 

73 self.name = f"gossiper-thread-{addr}" 

74 return addr

81 """Start the gossiper thread.""" 

82 logger.info(self.addr, "🏁 Starting gossiper...") 

83 return super().start()

86 """Stop the gossiper thread."""

88 self.__gossip_terminate_flag.set()

91 # Gossip 

92 ### 

93 

94 def add_message(self, msg: node_pb2.RootMessage) -> None: 

95 """ 

96 Add message to pending. 

97 

98 Args: 

99 msg: Message to send. 

100 pending_neis: Neighbors to send the message. 

101 

102 """ 

103 self.__pending_msgs_lock.acquire() 

104 pending_neis = [v[0] for addr, v in self.__neighbors.get_all(only_direct=True).items() if addr != self.addr and addr != msg.source] 

105 self.__pending_msgs.append((msg, pending_neis)) 

106 self.__pending_msgs_lock.release() 

107 

108 def check_and_set_processed(self, msg: node_pb2.RootMessage) -> bool: 

109 """ 

110 Check if message was already processed and set it as processed. 

111 

112 Args: 

113 msg: Message to check. 

114 

115 """ 

116 # If self address, return False 

117 if msg.source == self.addr: 

118 return False 

119 

120 # Check if message was already processed 

121 with self.__processed_messages_lock: 

122 if msg.gossip_message.hash in self.__processed_messages: 

123 return False 

124 # If there are more than X messages, remove the oldest one 

125 if len(self.__processed_messages) > Settings.gossip.AMOUNT_LAST_MESSAGES_SAVED: 

126 self.__processed_messages.pop(0) 

127 # Add message 

128 self.__processed_messages.append(msg.gossip_message.hash) 

129 return True 

130 

131 def run(self) -> None: 

132 """Run the gossiper thread.""" 

133 while not self.__gossip_terminate_flag.is_set(): 

134 t = time.time() 

135 messages_to_send = [] 

136 messages_left = self.messages_per_period 

137 

138 # Lock 

139 with self.__pending_msgs_lock: 

140 # Select the max amount of messages to send 

141 while messages_left > 0 and len(self.__pending_msgs) > 0: 

142 head_msg = self.__pending_msgs[0] 

143 # Select msgs 

144 if len(head_msg[1]) < messages_left: 

145 # Select all 

146 messages_to_send.append(head_msg) 

147 # Remove from pending 

148 self.__pending_msgs.pop(0) 

149 else: 

150 # Select only the first neis 

151 messages_to_send.append((head_msg[0], head_msg[1][:messages_left])) 

152 # Remove from pending 

153 self.__pending_msgs[0] = ( 

154 self.__pending_msgs[0][0], 

155 self.__pending_msgs[0][1][messages_left:], 

156 ) 

157 

158 # Send messages 

159 for msg, neis in messages_to_send: 

160 for nei in neis: 

161 nei.send(msg) 

162 # Sleep to allow periodicity 

163 sleep_time = max(0, self.period - (t - time.time())) 

164 time.sleep(sleep_time)

167 # Gossip Model (syncronous gossip not as a thread) 

168 ### 

169 

170 def gossip_weights( 

171 self, 

172 early_stopping_fn: Callable[[], bool], 

173 get_candidates_fn: Callable[[], list[str]], 

174 status_fn: Callable[[], Any], 

175 model_fn: Callable[[str], tuple[Any, str, int, list[str]]], # TODO: this can be simplified 

176 period: float, 

177 temporal_connection: bool, 

178 ) -> None: 

179 """ 

180 Gossip model weights. This is a synchronous gossip. End when there are no more neighbors to gossip. 

181 

182 Args: 

183 early_stopping_fn: Function to check if the gossip should stop. 

184 get_candidates_fn: Function to get the neighbors to gossip. 

185 status_fn: Function to get the status of the node. 

186 model_fn: Function to get the model of a neighbor. 

187 period: Period of gossip. 

188 temporal_connection: Flag to create a connection if neis not connected directly. 

189 

190 """ 

191 # Initialize list with status of nodes in the last X iterations 

192 last_x_status: list[Any] = [] 

193 j = 0 

194 

195 while True: 

196 # Get time to calculate frequency 

197 t = time.time() 

198 

199 # If the trainning has been interrupted, stop waiting 

200 if early_stopping_fn(): 

201 logger.info(self.addr, "Stopping model gossip process.") 

202 return 

203 

204 # Get nodes wich need models 

205 neis = get_candidates_fn() 

206 

207 # Determine end of gossip 

208 if neis == []: 

209 logger.info(self.addr, "🤫 Gossip finished.") 

210 return 

211 

212 # Save state of neighbors. If nodes are not responding gossip will stop 

213 logger.debug(self.addr, f"👥 Gossip remaining nodes: {neis}") 

214 if len(last_x_status) != Settings.gossip.EXIT_ON_X_EQUAL_ROUNDS: 

215 last_x_status.append(status_fn()) 

216 else: 

217 last_x_status[j] = str(status_fn()) 

218 j = (j + 1) % Settings.gossip.EXIT_ON_X_EQUAL_ROUNDS 

219 

220 # Check if las messages are the same 

221 for i in range(len(last_x_status) - 1): 

222 if last_x_status[i] != last_x_status[i + 1]: 

223 break 

224 logger.info( 

225 self.addr, 

226 f"⏹️ Gossiping exited for {Settings.gossip.EXIT_ON_X_EQUAL_ROUNDS} equal rounds.", 

227 ) 

228 logger.debug(self.addr, f"Gossip last status: {last_x_status[-1]}") 

229 return 

230 

231 # Select a random subset of neighbors 

232 samples = min(Settings.gossip.MODELS_PER_ROUND, len(neis)) 

233 neis = random.sample(neis, samples) 

234 # Getting all nodes and forcing tmp direct message 

235 neis_clients = [v[0] for k, v in self.__neighbors.get_all(only_direct=False).items() if k in neis] 

236 

237 # Generate and Send Model Partial Aggregations (model, node_contributors) 

238 for client in neis_clients: 

239 # Get Model 

240 model, command_name, round, model_hashes = model_fn(client.nei_addr) 

241 if model is None: 

242 continue 

243 

244 # Pre send weights 

245 presend_msg = self.build_msg_fn(PreSendModelCommand.get_name(), [command_name] + model_hashes, round, direct=True) 

246 presend_response = client.send(presend_msg, temporal_connection=temporal_connection) 

247 

248 # Send model 

249 if presend_response != "true": 

250 logger.debug( 

251 self.addr, f"Avoiding concurrent model sending to {client.nei_addr}. Msg: {command_name} | Hash: {model_hashes}" 

252 ) 

253 continue 

254 

255 # Send 

256 logger.debug(self.addr, f"🗣️ Gossiping model to {client.nei_addr}.") 

257 client.send(model, temporal_connection=temporal_connection) 

258 

259 # Sleep to allow periodicity 

260 sleep_time = max(0, period - (t - time.time())) 

261 time.sleep(sleep_time)
