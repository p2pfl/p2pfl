44def set_standalone_settings(disable_ray: bool = False) -> None: 

45 """ 

46 Set settings for testing. 

47 

48 Important: 

49 - HEARTBEAT_PERIOD: Too high values can cause late node discovery/fault detection. Too low values can cause high CPU usage. 

50 - GOSSIP_PERIOD: Too low values can cause high CPU usage. 

51 - TTL: Low TTLs can cause that some messages are not delivered. 

52 

53 """ 

54 Settings.GRPC_TIMEOUT = 0.5 

55 Settings.HEARTBEAT_PERIOD = 5 

56 Settings.HEARTBEAT_TIMEOUT = 40 

57 Settings.GOSSIP_PERIOD = 1 

58 Settings.TTL = 40 

59 Settings.GOSSIP_MESSAGES_PER_PERIOD = 9999999999 

60 Settings.AMOUNT_LAST_MESSAGES_SAVED = 10000 

61 Settings.GOSSIP_MODELS_PERIOD = 1 

62 Settings.GOSSIP_MODELS_PER_ROUND = 4 

63 Settings.GOSSIP_EXIT_ON_X_EQUAL_ROUNDS = 10 

64 Settings.TRAIN_SET_SIZE = 4 

65 Settings.VOTE_TIMEOUT = 60 

66 Settings.AGGREGATION_TIMEOUT = 60 

67 Settings.WAIT_HEARTBEATS_CONVERGENCE = 0.2 * Settings.HEARTBEAT_TIMEOUT 

68 Settings.LOG_LEVEL = "INFO" 

69 Settings.EXCLUDE_BEAT_LOGS = True 

70 Settings.DISABLE_RAY = disable_ray 

71 logger.set_level(Settings.LOG_LEVEL) # Refresh (maybe already initialized)

74def __parse_args() -> argparse.Namespace: 

75 parser = argparse.ArgumentParser(description="P2PFL MNIST experiment using the Web Logger.") 

76 parser.add_argument("--nodes", type=int, help="The number of nodes.", default=2) 

77 parser.add_argument("--rounds", type=int, help="The number of rounds.", default=2) 

78 parser.add_argument("--epochs", type=int, help="The number of epochs.", default=1) 

79 parser.add_argument("--show_metrics", action="store_true", help="Show metrics.", default=True) 

80 parser.add_argument("--measure_time", action="store_true", help="Measure time.", default=False) 

81 parser.add_argument("--token", type=str, help="The API token for the Web Logger.", default="") 

82 parser.add_argument("--protocol", type=str, help="The protocol to use.", default="grpc", choices=["grpc", "unix", "memory"]) 

83 parser.add_argument("--framework", type=str, help="The framework to use.", default="pytorch", choices=["pytorch", "tensorflow", "flax"]) 

84 parser.add_argument("--aggregator", type=str, help="The aggregator to use.", default="fedavg", choices=["fedavg", "scaffold"]) 

85 parser.add_argument("--profiling", action="store_true", help="Enable profiling.", default=False) 

86 parser.add_argument("--reduced_dataset", action="store_true", help="Use a reduced dataset just for testing.", default=False) 

87 parser.add_argument("--use_scaffold", action="store_true", help="Use the Scaffold aggregator.", default=False) 

88 parser.add_argument("--disable_ray", action="store_true", help="Disable Ray.", default=False) 

89 parser.add_argument( 

90 "--topology", 

91 type=str, 

92 choices=[t.value for t in TopologyType], 

93 default="line", 

94 help="The network topology (star, full, line, ring).", 

95 ) 

96 args = parser.parse_args() 

97 # parse topology to TopologyType enum 

98 args.topology = TopologyType(args.topology) 

99 

100 return args

122def mnist( 

123 n: int, 

124 r: int, 

125 e: int, 

126 show_metrics: bool = True, 

127 measure_time: bool = False, 

128 protocol: str = "grpc", 

129 framework: str = "pytorch", 

130 aggregator: str = "fedavg", 

131 reduced_dataset: bool = False, 

132 topology: TopologyType = TopologyType.LINE, 

133) -> None: 

134 """ 

135 P2PFL MNIST experiment. 

136 

137 Args: 

138 n: The number of nodes. 

139 r: The number of rounds. 

140 e: The number of epochs. 

141 show_metrics: Show metrics. 

142 measure_time: Measure time. 

143 protocol: The protocol to use. 

144 framework: The framework to use. 

145 aggregator: The aggregator to use. 

146 reduced_dataset: Use a reduced dataset just for testing. 

147 topology: The network topology (star, full, line, ring). 

148 

149 """ 

150 if measure_time: 

151 start_time = time.time() 

152 

153 # Check settings 

154 if n > Settings.TTL: 

155 raise ValueError( 

156 "For in-line topology TTL must be greater than the number of nodes." "Otherwise, some messages will not be delivered." 

157 ) 

158 

159 # Imports 

160 if framework == "tensorflow": 

161 from p2pfl.learning.frameworks.tensorflow.keras_learner import KerasLearner 

162 

163 model_fn = create_tensorflow_model 

164 learner = KerasLearner 

165 elif framework == "pytorch": 

166 from p2pfl.learning.frameworks.pytorch.lightning_learner import LightningLearner 

167 

168 model_fn = create_pytorch_model 

169 learner = LightningLearner # type: ignore 

170 else: 

171 raise ValueError(f"Framework {args.framework} not added on this example.") 

172 

173 # Data 

174 data = P2PFLDataset.from_huggingface("p2pfl/MNIST") 

175 partitions = data.generate_partitions( 

176 n * 50 if reduced_dataset else n, 

177 RandomIIDPartitionStrategy, # type: ignore 

178 ) 

179 

180 # Node Creation 

181 nodes = [] 

182 for i in range(n): 

183 address = f"node-{i}" if protocol == "memory" else f"unix:///tmp/p2pfl-{i}.sock" if protocol == "unix" else "127.0.0.1" 

184 

185 # Nodes 

186 node = Node( 

187 model_fn(), 

188 partitions[i], 

189 learner=learner, # type: ignore 

190 protocol=InMemoryCommunicationProtocol if protocol == "memory" else GrpcCommunicationProtocol, # type: ignore 

191 address=address, 

192 simulation=True, 

193 aggregator=Scaffold() if aggregator == "scaffold" else None, 

194 ) 

195 node.start() 

196 nodes.append(node) 

197 

198 try: 

199 adjacency_matrix = TopologyFactory.generate_matrix(topology, len(nodes)) 

200 TopologyFactory.connect_nodes(adjacency_matrix, nodes) 

201 

202 wait_convergence(nodes, n - 1, only_direct=False, wait=60) # type: ignore 

203 

204 if r < 1: 

205 raise ValueError("Skipping training, amount of round is less than 1") 

206 

207 # Start Learning 

208 nodes[0].set_start_learning(rounds=r, epochs=e) 

209 

210 # Wait and check 

211 wait_to_finish(nodes, timeout=60 * 60) # 1 hour 

212 

213 # Local Logs 

214 if show_metrics: 

215 local_logs = logger.get_local_logs() 

216 if local_logs != {}: 

217 logs_l = list(local_logs.items())[0][1] 

218 # Plot experiment metrics 

219 for round_num, round_metrics in logs_l.items(): 

220 for node_name, node_metrics in round_metrics.items(): 

221 for metric, values in node_metrics.items(): 

222 x, y = zip(*values) 

223 plt.plot(x, y, label=metric) 

224 # Add a red point to the last data point 

225 plt.scatter(x[-1], y[-1], color="red") 

226 plt.title(f"Round {round_num} - {node_name}") 

227 plt.xlabel("Epoch") 

228 plt.ylabel(metric) 

229 plt.legend() 

230 plt.show() 

231 

232 # Global Logs 

233 global_logs = logger.get_global_logs() 

234 if global_logs != {}: 

235 logs_g = list(global_logs.items())[0][1] # Accessing the nested dictionary directly 

236 # Plot experiment metrics 

237 for node_name, node_metrics in logs_g.items(): 

238 for metric, values in node_metrics.items(): 

239 x, y = zip(*values) 

240 plt.plot(x, y, label=metric) 

241 # Add a red point to the last data point 

242 plt.scatter(x[-1], y[-1], color="red") 

243 plt.title(f"{node_name} - {metric}") 

244 plt.xlabel("Epoch") 

245 plt.ylabel(metric) 

246 plt.legend() 

247 plt.show() 

248 except Exception as e: 

249 raise e 

250 finally: 

251 # Stop Nodes 

252 for node in nodes: 

253 node.stop() 

254 

255 if measure_time: 

256 print("--- %s seconds ---" % (time.time() - start_time)) 

257 

258 

259if __name__ == "__main__": 

260 # Parse args 

261 args = __parse_args() 

262 

263 set_standalone_settings(disable_ray=args.disable_ray) # todo: not working on the logger because it is imported at the top of the file 

264 

265 if args.profiling: 

266 import os # noqa: I001 

267 import yappi # type: ignore 

268 

269 # Start profiler 

270 yappi.start() 

271 

272 # Set logger 

273 if args.token != "": 

274 logger.connect_web("http://localhost:3000/api/v1", args.token) 

275 

276 # Launch experiment 

277 try: 

278 mnist( 

279 args.nodes, 

280 args.rounds, 

281 args.epochs, 

282 show_metrics=args.show_metrics, 

283 measure_time=args.measure_time, 

284 protocol=args.protocol, 

285 framework=args.framework, 

286 aggregator=args.aggregator, 

287 reduced_dataset=args.reduced_dataset, 

288 topology=args.topology, 

289 ) 

290 finally: 

291 if args.profiling: 

292 # Stop profiler 

293 yappi.stop() 

294 # Save stats 

295 profile_dir = os.path.join("profile", "mnist", str(uuid.uuid4())) 

296 os.makedirs(profile_dir, exist_ok=True) 

297 for thread in yappi.get_thread_stats(): 

298 yappi.get_func_stats(ctx_id=thread.id).save(f"{profile_dir}/{thread.name}-{thread.id}.pstat", type="pstat")
